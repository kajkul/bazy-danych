
1.1. 
    SELECT c.CLIENT_ID FROM CLIENT AS c LEFT JOIN `ORDER` as o ON c.CLIENT_ID = o.CLIENT_ID WHERE o.ORDER_ID IS NULL GROUP BY c.CLIENT_ID;
    SELECT c.CLIENT_ID FROM CLIENT WHERE CLIENT_ID NOT IN (SELECT DISTINCT CLIENT_ID FROM `ORDER`);

1.2. 
    SELECT DISTINCT u.UNIT_ID, u.UNIT_NAME FROM INVENTORY AS i RIGHT JOIN UNIT_OF_MEASUREMENT as u ON u.UNIT_ID = i.UNIT_ID WHERE i.PRODUCT_ID IS NULL;

1.3. 
    SELECT prod.PRODUCT_ID, prod.PRODUCT_NAME, AVG(pos.SELL_PRICE - prod.PURCHASE_PRICE) as markup FROM ORDER_POSITION AS pos LEFT JOIN PRODUCT AS prod ON prod.PRODUCT_ID = pos.PRODUCT_ID GROUP BY prod.PRODUCT_ID, prod.PRODUCT_NAME ORDER BY markup DESC;

2.1. 
    SELECT DISTINCT d.DEPARTMENT_NAME, o.ORDER_ID FROM DEPARTMENT as d
    LEFT JOIN EMPLOYEE as e ON e.DEPARTMENT_ID = d.DEPARTMENT_ID 
    INNER JOIN `ORDER` AS o ON o.EMPLOYEE_ID = e.EMPLOYEE_NAME;

2.2.
   SELECT d.DEPARTMENT_ID, d.DEPARTMENT_NAME, SUM(p.SELL_PRICE * p.AMOUNT) 
   FROM DEPARTMENT as d 
   LEFT JOIN EMPLOYEE as e ON e.DEPARTMENT_ID = d.DEPARTMENT_ID 
   INNER JOIN `ORDER` AS o ON o.EMPLOYEE_ID = e.EMPLOYEE_NAME
   LEFT JOIN ORDER_POSITION AS p ON p.ORDER_ID = o.ORDER_ID
   GROUP BY d.DEPARTMENT_ID, d.DEPARTMENT_NAME;
 
2.3. 
    SELECT d.DEPARTMENT_ID, d.DEPARTMENT_NAME, os.STATUS_NAME, SUM(p.SELL_PRICE * p.AMOUNT)
    FROM DEPARTMENT AS d 
    LEFT JOIN EMPLOYEE AS e ON e.DEPARTMENT_ID = d.DEPARTMENT_ID 
    INNER JOIN `ORDER` AS o ON o. EMPLOYEE_ID = e.EMPLOYEE_NAME
    LEFT JOIN ORDER_POSITION AS p ON p.ORDER_ID = o.ORDER_ID
    LEFT JOIN ORDER_STATUS AS os ON os.STATUS_ID = o.ORDER_STATUS
    GROUP BY d.DEPARTMENT_ID, DEPARTMENT_NAME, os.STATUS_NAME;

3.1. SELECT CONCAT(PRODUCT_NAME, " (", DESCRIPTION, ")") FROM PRODUCT;
3.2. SELECT CONCAT_WS(",", PRODUCT_ID, CATEGORY_ID, PRODUCT_NAME, PURCHASE_PRICE, DESCRIPTION) FROM PRODUCT;
3.3. SELECT DEPARTMENT_ID, GROUP_CONCAT(CONCAT(FIRST_NAME, " ", LAST_NAME) SEPARATOR ", ") FROM EMPLOYEE GROUP BY DEPARTMENT_ID;

Wykorzystując wbudowaną funkcję GROUP_CONCAT wyświetl nazwę kategorii, a w
kolejnej kolumnie listę nazw produktów z tej kategorii, które nigdy nie były sprzedane.

3.4. SELECT CATEGORY_NAME, GROUP_CONCAT(PRODUCT_NAME SEPARATOR ", ") FROM CATEGORY NATURAL JOIN PRODUCT GROUP BY CATEGORY_ID, CATEGORY_NAME;

4.1. SELECT ORDER_ID, IF(ORDER_STATUS IN (3,4,7,10), "Zakończone", "Inny") FROM `ORDER`;

4.2.
    SELECT 
    PRODUCT_ID,
    CASE 
        WHEN AMOUNT = 0 THEN 'Brak'
        WHEN AMOUNT < 10 THEN 'Mala iloss'
        WHEN AMOUNT <= 30 THEN 'Srednia ilosc'
        WHEN AMOUNT > 30 THEN 'Duza ilosc'
    END AS dostepnosc
    FROM INVENTORY;

4.3. SELECT CLIENT_ID, IFNULL(TAX_IDENTIFIER, 'Klient indywidualny') FROM CLIENT;
