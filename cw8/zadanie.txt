1.1. 
    SELECT
	CLIENT.FULL_NAME,
	CLIENT_ADDRESS.STREET,
	LENGTH(CLIENT_ADDRESS.STREET),
	SUBSTR(CLIENT_ADDRESS.STREET, 1, 5)
    FROM CLIENT
    LEFT JOIN CLIENT_ADDRESS ON CLIENT.CLIENT_ID = CLIENT_ADDRESS.CLIENT_ID;

1.2. 
    SELECT
	CLIENT.SHORT_NAME,
	ADDRESS_TYPE_NAME
    FROM CLIENT
    LEFT JOIN CLIENT_ADDRESS ON CLIENT.CLIENT_ID = CLIENT_ADDRESS.CLIENT_ID
    NATURAL JOIN ADDRESS_TYPE;

1.3.
    SELECT
	PRODUCT.PRODUCT_NAME,
	CATEGORY.CATEGORY_NAME,
	SUBSTR(PRODUCT.PRODUCT_NAME, 1, 3)
    FROM PRODUCT
    JOIN CATEGORY ON PRODUCT.CATEGORY_ID = CATEGORY.CATEGORY_ID;

1.4.
    SELECT
	PRODUCT.PRODUCT_NAME,
	PRODUCT.DESCRIPTION
    FROM PRODUCT
    JOIN ORDER_POSITION ON PRODUCT.PRODUCT_ID = ORDER_POSITION.PRODUCT_ID
    GROUP BY 
	PRODUCT.PRODUCT_ID,
	PRODUCT.PRODUCT_NAME,
	PRODUCT.DESCRIPTION;

2.1.
    SELECT
	c.FULL_NAME,
	o.ORDER_NUMBER,
	o.ORDER_DATE,
	DATEDIFF(CURDATE(), o.ORDER_DATE)
    FROM `ORDER` AS o
    JOIN CLIENT AS c ON o.CLIENT_ID = c.CLIENT_ID;

2.2.
    SELECT
	o.ORDER_ID,
	o.ORDER_DATE,
	DATE_ADD(o.ORDER_DATE, INTERVAL 14 DAY)
    FROM `ORDER` AS o
    WHERE o.ORDER_STATUS NOT IN (3, 7, 10);

2.3. 
    SELECT
	e.FIRST_NAME,
	e.LAST_NAME,
	e.DATE_OF_EMPLOYMENT,
	DAYNAME(e.DATE_OF_EMPLOYMENT),
	d.DEPARTMENT_NAME
    FROM EMPLOYEE AS e
    LEFT JOIN DEPARTMENT AS d ON e.DEPARTMENT_ID = d.DEPARTMENT_ID;

2.4.
SELECT
    c.FULL_NAME,
    (DATEDIFF(MAX(o.ORDER_DATE), MIN(o.ORDER_DATE))),
FROM CLIENT AS c
LEFT JOIN `ORDER` AS o ON c.CLIENT_ID = o.CLIENT_ID
GROUP BY 
    c.CLIENT_ID,
    c.FULL_NAME;

3.1.
    SELECT
	c.SHORT_NAME,
	GROUP_CONCAT(a.STREET SEPARATOR ', ')
    FROM CLIENT AS c
    LEFT JOIN CLIENT_ADDRESS AS a ON c.CLIENT_ID = a.CLIENT_ID
    GROUP BY 
	c.CLIENT_ID,
	c.SHORT_NAME;

3.2.
    SELECT
	CITY,
	GROUP_CONCAT(DISTINCT SUBSTR(POSTAL_CODE, 1, 2) SEPARATOR ', ') AS postal
    FROM CLIENT_ADDRESS
    GROUP BY CITY;

3.3.
    SELECT
	CONCAT(e.LAST_NAME, ' ', e.FIRST_NAME),
	d.DEPARTMENT_NAME,
	DATEDIFF(CURDATE(), e.DATE_OF_EMPLOYMENT)
    FROM EMPLOYEE AS e
    LEFT JOIN DEPARTMENT AS d ON e.DEPARTMENT_ID = d.DEPARTMENT_ID;

3.4.
    SELECT
	CONCAT(c.FULL_NAME, ' - ', c.SHORT_NAME) AS name_display,
	GROUP_CONCAT(RIGHT(o.ORDER_NUMBER, 3) SEPARATOR ', ') AS order_suffixes
    FROM CLIENT AS c
    LEFT JOIN `ORDER` AS o ON c.CLIENT_ID = o.CLIENT_ID
    GROUP BY 
	c.CLIENT_ID,
	c.FULL_NAME,
	c.SHORT_NAME;

3.5.
    SELECT
	e.FIRST_NAME,
	e.LAST_NAME,
	CONCAT(
	    LOWER(SUBSTR(e.FIRST_NAME, 1, 3)),
	    UPPER(RIGHT(e.LAST_NAME, 4)),
	    LENGTH(e.FIRST_NAME) + LENGTH(a.STREET)
	) AS password
    FROM EMPLOYEE AS e
    JOIN CLIENT_ADDRESS AS a ON e.EMPLOYEE_NAME = a.CLIENT_ID;
